TERMINAL 1 (Setup + Supervisor)
cd OS-Jackfruit
mkdir rootfs-base
tar -xzf alpine-minirootfs-*.tar.gz -C rootfs-base
cp -a rootfs-base rootfs-alpha
cp -a rootfs-base rootfs-beta


cd OS-Jackfruit/boilerplate
make clean
make
sudo insmod monitor.ko
lsmod | grep monitor
ls -l /dev/container_monitor
sudo dmesg -C
sudo ./engine supervisor ../rootfs-base

TERMINAL 2 (Tasks)

cd OS-Jackfruit/boilerplate

TASK 1 & 7: Run workloads
sudo ./engine run test ../rootfs-alpha /bin/sh

sudo ./engine start cpu ../rootfs-alpha ./cpu_hog
sudo ./engine start io ../rootfs-beta ./io_pulse

TASK 2: List containers
sudo ./engine ps

TASK 3: View logs
sudo ./engine logs cpu

TASK 4: Run one-off command
sudo ./engine run test ../rootfs-alpha /bin/ls


TASK 5 & 6: Memory monitor test
sudo dmesg -C
sudo ./engine start mem ../rootfs-alpha ./memory_hog
sleep 2
sudo dmesg | tail -20


TASK 8: Process check + multiple containers
sudo ./engine start alpha ../rootfs-alpha /bin/ls
sudo ./engine start beta ../rootfs-beta /bin/ls
sudo ./engine ps

ps aux | grep engine
sudo pkill -9 engine
sudo kill -9 <pid>
ps aux | grep engine

FINAL: CI Test
make ci
